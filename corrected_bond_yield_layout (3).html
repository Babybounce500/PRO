Effektive Rendite der Anleihe: -

// Hier können Sie eine Formel für die Anleiheberechnung basierend auf den oben genannten Kennzahlen hinzufügen. // Als einfaches Beispiel für die effektive Rendite: let nennwert = parseFloat(document.getElementById("nennwert").value); let laufzeit = parseFloat(document.getElementById("laufzeit").value); let zinssatz = parseFloat(document.getElementById("zinssatz").value); let emissionspreis = parseFloat(document.getElementById("emissionspreis").value); let rueckzahlungspreis = parseFloat(document.getElementById("rueckzahlungspreis").value); let zinszahlungsfrequenz = parseFloat(document.getElementById("zinszahlungsfrequenz").value); let spread = parseFloat(document.getElementById("spread").value); let zinszahlung = nennwert * zinssatz / 100; let durchschnittlicherJahresertrag = (zinszahlung * zinszahlungsfrequenz + (rueckzahlungspreis - emissionspreis) / laufzeit); let durchschnittlicherJahrespreis = (emissionspreis + rueckzahlungspreis) / 2; let effektiveRendite = durchschnittlicherJahresertrag / durchschnittlicherJahrespreis * 100 + spread; document.getElementById("yield").innerText = effektiveRendite.toFixed(2) + "%"; }
